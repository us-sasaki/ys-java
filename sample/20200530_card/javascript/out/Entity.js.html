/**
 * Entity クラス定義
 * Entity は canvas 上に描画されるオブジェクトを表します。
 */
/*-------------------------
 * class def & constructor
 */
var Entity = function() {


/*----------------
 * instance field
 */
	this.setParent(undefined);
	this.setSize(0, 0);
	this.setPosition(0, 0);
	this.setDirection(0); // upright
	this.isVisible = true;
}
/*------------------
 * static constants
 */
	Entity.UPRIGHT = 0; // 下のプレイヤーから見て直立
	Entity.LEFT_VIEW = 1; // 左のプレイヤーから見て直立
	Entity.UPSIDE_DOWN = 2; // 上のプレイヤーから見て直立
	Entity.RIGHT_VIEW = 3; // 右のプレイヤーから見て直立

/*------------------
 * instance methods
 */
	/**
	 * 親オブジェクト(Entities)を設定します。
	 * このメソッドは親オブジェクトからのみ呼ばれます。
	 */
	Entity.prototype.setParent = function(parent) {
		this.parent = parent;
	}
	Entity.prototype.setSize = function(w, h) {
		this.w = w;
		this.h = h;
	}
	/**
	 * 位置を指定します。左上が(0,0)で右下座標系です。
	 */
	Entity.prototype.setPosition = function(x, y) {
		this.x = x;
		this.y = y;
	}
	
	/**
	 * この Entity の向き(0-3)を指定します。
	 */
	Entity.prototype.setDirection = function(dir) {
		if (dir!=0 && dir!=1 && dir!=2 && dir!=3) return;
		this.direction = dir;
	}
	
	/**
	 * この Entity の向き(0-3)を返却します。
	 */
	Entity.prototype.getDirection = function() {
		return this.direction;
	}
	
	/**
	 * この Entity の位置、大きさを取得します。
	 *
	 * @return		{x: <<x座標>>, y: <<y座標>>, w: <<幅>>, h: <<高さ>>
	 */
	Entity.prototype.getRect = function() {
		return {
			x: this.x,
			y: this.y,
			w: this.w,
			h: this.h };
	}
	
	/**
	 * この Entity を描画します。
	 * このメソッドは親オブジェクトからのみ呼ばれます。
	 * 子クラスでは isVisible が false のとき描画しないようにする必要があります
	 */
	Entity.prototype.draw = function(ctx) {
		throw new Error("not implemented");
	}
